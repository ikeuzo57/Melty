# To initialize

> yarn
> git submodule init

# To run

> yarn && yarn watch (takes a minute or so)
> Build with cmd+shift+b (or go to command palette and run build task)
> ./scripts/code.sh

# To install latest spectacular

> ./scripts/build-spectacular.sh

# To package

> yarn gulp vscode-darwin-arm64
